```typescript
const loadPatientInfo = async (patientId: number) => {
  try {
    console.log('[loadPatientInfo] 开始加载患者信息，patientId:', patientId)
    const patient: any = await getInpatientPatient(patientId)
    
    console.log('[loadPatientInfo] 患者信息响应:', patient)
    
    if (patient) {
      const patientFields = [
        'patientName', 'gender', 'age', 'inpatientNo', 
        'department', 'bedNo', 'admissionDate', 'caseNo'
      ]
      
      patientFields.forEach(field => {
        let value: any
        switch (field) {
          case 'patientName':
            value = patient.name || ''
            break
          // ... 其他字段
          case 'caseNo':
            value = ''  // ✅ 不从患者信息中获取，让后端生成
            break
          default:
            value = ''
        }
        ;(formData as any)[field] = value
      })
      
      patientInfoLoaded.value = true
      console.log('[loadPatientInfo] 患者信息加载完成')
      await nextTick()
    } else {
      console.warn('[loadPatientInfo] 未找到患者信息')
      ElMessage.warning('未找到患者信息')
      patientInfoLoaded.value = true  // ✅ 避免一直loading
    }
  } catch (error: any) {
    console.error('[loadPatientInfo] 加载患者信息失败:', error)
    ElMessage.warning('加载患者信息失败')
    patientInfoLoaded.value = true  // ✅ 避免一直loading
  }
}
```

**检查项**:
- ✅ `caseNo` 设置为空字符串（不从患者信息获取）
- ✅ 成功/失败都设置 `patientInfoLoaded = true`
- ✅ 详细的日志输出

```typescript
const loadRecordData = () => {
  if (props.recordData?.content) {
    try {
      const content = typeof props.recordData.content === 'string' 
        ? JSON.parse(props.recordData.content) 
        : props.recordData.content
      
      const progressRecord = content.progressRecord || {}
      
      // 定义不应该被病案数据覆盖的患者基本信息字段
      const patientInfoFields = [
        'patientName', 'gender', 'age', 'inpatientNo', 
        'department', 'bedNo', 'admissionDate', 'caseNo'
      ]
      
      // 填充病案特有字段（排除患者基本信息）
      Object.keys(progressRecord).forEach(key => {
        if (!patientInfoFields.includes(key) && progressRecord[key] !== undefined) {
          (formData as any)[key] = progressRecord[key]
        }
      })
      
      // 同时更新顶层字段（recordDateTime, status等）
      if (props.recordData.recordDateTime) {
        formData.recordDateTime = props.recordData.recordDateTime
      }
      if (props.recordData.status) {
        formData.status = props.recordData.status
      }
      
      console.log('病案数据加载完成')
    } catch (error) {
      console.error('Failed to parse record content:', error)
    }
  }
}
```

**检查项**:
- ✅ 解析 JSON 格式的 content
- ✅ **保护患者基本信息字段不被覆盖**
- ✅ 只更新病案特有字段
- ✅ 同时更新顶层字段

```typescript
const handleSubmit = async () => {
  if (!formRef.value) return

  await formRef.value.validate(async (valid) => {
    if (!valid) return

    submitting.value = true
    try {
      const recordContent = {
        progressRecord: {
          recordDateTime: formData.recordDateTime,
          caseFeatures: formData.caseFeatures,
          preliminaryDiagnosis: formData.preliminaryDiagnosis,
          preliminaryDiagnosisCode: formData.preliminaryDiagnosisCode,
          diagnosisBasis: formData.diagnosisBasis,
          differentialDiagnosis: formData.differentialDiagnosis,
          treatmentPlan: formData.treatmentPlan,
          precautions: formData.precautions,
          doctorSignature: formData.doctorSignature,
        },
      }

      // 编辑模式：只更新必要字段，避免触发唯一约束
      const submitData = isEdit.value 
        ? {
            recordType: 'progress',
            content: JSON.stringify(recordContent),
            status: formData.status,
          }
        : {
            patientId: formData.patientId || props.patientId,
            caseNo: formData.caseNo,  // ✅ 空字符串，后端会生成
            recordType: 'progress',
            content: JSON.stringify(recordContent),
            status: formData.status,
          }

      console.log('提交数据:', submitData)

      if (isEdit.value) {
        await updateInpatientRecord(formData.id!, submitData)
        ElMessage.success('更新成功')
      } else {
        await createInpatientRecord(submitData)
        ElMessage.success('创建成功')
      }
      handleClose()
      emit('success')
    } catch (error: any) {
      console.error('提交失败:', error)
      console.error('错误详情:', error.response?.data)
      ElMessage.error(error.message || '操作失败')
    } finally {
      submitting.value = false
    }
  })
}
```

```typescript
export const createInpatientRecord = async (req: AuthRequest, res: Response) => {
  try {
    const currentUser = req.user!
    
    // 确保doctorId被设置（数据隔离）
    req.body.doctorId = currentUser.id
    
    // ✅ 如果没有提供caseNo，自动生成
    if (!req.body.caseNo || req.body.caseNo.trim() === '') {
      // 生成格式：CASE + 时间戳(6位) + 随机数(3位)
      const timestamp = Date.now().toString().slice(-6)
      const random = Math.floor(Math.random() * 1000).toString().padStart(3, '0')
      req.body.caseNo = `CASE${timestamp}${random}`
      console.log('✅ 自动生成病案号:', req.body.caseNo)
    } else {
      console.log('ℹ️ 使用前端提供的病案号:', req.body.caseNo)
    }
    
    // 序列化content字段为JSON字符串
    if (req.body.content && typeof req.body.content === 'object') {
      req.body.content = JSON.stringify(req.body.content)
    }
    
    console.log('创建病案 - 请求体:', req.body)
    
    const record = await InpatientRecord.create(req.body)
    res.json({
      code: 200,
      data: record,
      message: '创建成功',
    })
  } catch (err: any) {
    console.error('Create record error:', err)
    console.error('Error details:', err.errors)
    return error(res, 500, err.message)
  }
}
```

**检查项**:
- ✅ 自动设置 `doctorId`

- ✅ 序列化 `content` 为 JSON 字符串

## ⚠️ 常见问题排查

### 问题1: 对话框打开但一直显示 loading

**可能原因**:
1. `patientId` 为空
2. `loadPatientInfo` 失败但没有设置 `patientInfoLoaded = true`

**解决方法**:
- 检查控制台日志，确认 `patientId` 是否有值
- 检查网络请求是否成功
- 确认代码中所有异常分支都设置了 `patientInfoLoaded = true`

---

### 问题2: 提交时提示 500 错误

**可能原因**:
1. `caseNo` 使用了已存在的住院号
2. 后端没有自动生成病案号的逻辑

**解决方法**:
- 确认前端 `caseNo` 是空字符串
- 确认后端有自动生成逻辑
- 检查后端控制台错误日志

---

### 问题3: 编辑时患者信息被覆盖

**可能原因**:
1. 没有先加载患者信息就加载病案数据
2. `loadRecordData` 没有保护患者基本信息字段

**解决方法**:
- 确认 watch 中先调用 `loadPatientInfo` 再调用 `loadRecordData`
- 确认 `loadRecordData` 中有 `patientInfoFields` 保护列表
